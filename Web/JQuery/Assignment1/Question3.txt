1)  //Selects all SPAN with class name: “message”
    $('span.message')

2)  //Selects first DIV in a document that has class: “box”
    $('div.box:first')

3)  //Selects all BUTTONS in a document. 
    $('button')

4)  //Selects all IMAGES having alternate text as: “hello”
    $('img[alt="hello"]')

5)  //Selects all TEXT BOXES under a div having Id = “myDiv”
    $('#mydiv').find('input[type="text"]')

6)  //Selects all TEXT BOXES with name starting by: “txt”
    $('input[type="text"][name^="txt"]')

7)  //Selects all PARAGRAPHS element, not having class “box”
    $('p').not('.box')

8)  //Selects all the DIVS having either “box” or “error” class.
    $('div.box,.error')

9)  //Selects all the DIVS having both “box” and “error” class.
    $("div .box.error")

10) //Find all the SPAN having “info” class in a DIV with id = “myDiv”
    $("div#myDiv span.info")
